Data Cleaning

claims_clean <- claims %>%
  filter(!(STATUS1 %in% "BILLED")) %>%
  select(PATIENTID, DEPARTMENTID, DIAGNOSIS1, SERVICEDATE, HEALTHCARECLAIMTYPEID1, HEALTHCARECLAIMTYPEID2) %>% 
  mutate(date_YYYYMM = year(SERVICEDATE)*100 +  month(SERVICEDATE))
